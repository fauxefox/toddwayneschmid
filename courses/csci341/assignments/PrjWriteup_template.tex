% These are custom commands in LaTeX called "macros". They allow you to write 
% shorter code. For example, to get the natural numbers "N", the code below
% allows you to write \NN instead of \mathbb{N}. Think about these as being 
% "global variables" like in python or Java. Add or remove macros at will.
\newcommand{\NN}{\mathbb{N}}                % 0, 1, 2, 3, 4, ...
\newcommand{\ZZ}{\mathbb{Z}}                % ..., -2, -1, 0, 1, 2, ...
\newcommand{\QQ}{\mathbb{Q}}                % n/m for n and m in \NN and m > 0
\newcommand{\RR}{\mathbb{R}}                % real numbers

% This macro I have written looks better than the \xrightarrow{label} code that
% gets used in the notes. You just need to use the code \tr{label} to get that 
% nice compact transition notation.
\newcommand{\tr}[1]{
    \mathrel{
        \raisebox{-1pt}{
            \(\xrightarrow{#1}\)
            }
        }
    }
